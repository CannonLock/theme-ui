export const Radii = () => {
  const { theme } = useThemeUI()
  if (typeof theme.radii === 'object') {
    return (
      <React.Fragment>
        <Divider />
        <h2>Borders</h2>
        <Flex sx={{ flexWrap: 'wrap' }}>
          {Object.keys(theme.radii).map((key) => (
            <Box
              key={`border_${key}`}
              p={3}
              m={2}
              color="text"
              bg="muted"
              sx={{
                border: '2px solid black',
                borderRadius: { key },
                minWidth: '100px',
              }}>
              {key}
            </Box>
          ))}
        </Flex>
      </React.Fragment>
    )
  }
  return null
}
